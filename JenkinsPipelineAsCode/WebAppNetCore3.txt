pipeline {
    agent none
    stages {
        stage('Pull Repo'){
            agent {
                node{
                    label ''
                }
                
            }
            steps{
				sh '''
				if [ -d .git ]; then
				  git status;
				  git pull origin master;
				else
				  
				  git clone https://github.com/jjw24/DevOpsTooling.git .;
				fi;
				'''
            }
        }
        stage('Docker build') {
            agent any
            steps {
                script{
                    docker.build ('webappnetcore3','./Docker/WebAppCore3/')
                }
            }
        }
    }
}
